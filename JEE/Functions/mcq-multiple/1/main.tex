\item If $y=f(x)=\frac{x+2}{x-1}$ then
\hfill{2008S}
\begin{enumerate}
    \item $x=f(y)$
    \item $f(1)=3$
    \item $y$ increase with $x$ for $x<1$
    \item $f$ is a rational function on $x$
\end{enumerate}
